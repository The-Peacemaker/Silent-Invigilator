import React, { useState, useRef, useEffect } from 'react';
import { Camera, AlertTriangle, CheckCircle, Eye, Users, Activity } from 'lucide-react';

const SilentInvigilator = () => {
  const [isMonitoring, setIsMonitoring] = useState(false);
  const [alerts, setAlerts] = useState([]);
  const [stats, setStats] = useState({
    suspiciousActions: 0,
    totalStudents: 1,
    alertLevel: 'normal'
  });
  const [behaviorMetrics, setBehaviorMetrics] = useState({
    headPose: 0,
    eyeGaze: 0,
    handMovement: 0,
    voiceDetection: 0,
    phoneDetection: 0,
    multiplePersons: 0
  });
  
  const videoRef = useRef(null);
  const canvasRef = useRef(null);
  const streamRef = useRef(null);
  const animationRef = useRef(null);

  useEffect(() => {
    return () => {
      if (streamRef.current) {
        streamRef.current.getTracks().forEach(track => track.stop());
      }
      if (animationRef.current) {
        cancelAnimationFrame(animationRef.current);
      }
    };
  }, []);

  const startMonitoring = async () => {
    try {
      const stream = await navigator.mediaDevices.getUserMedia({ 
        video: { width: 640, height: 480 },
        audio: true 
      });
      
      if (videoRef.current) {
        videoRef.current.srcObject = stream;
        streamRef.current = stream;
        setIsMonitoring(true);
        processVideo();
        simulateBehaviorAnalysis();
      }
    } catch (err) {
      addAlert('Camera access denied. Please enable camera permissions.', 'error');
    }
  };

  const stopMonitoring = () => {
    if (streamRef.current) {
      streamRef.current.getTracks().forEach(track => track.stop());
    }
    if (animationRef.current) {
      cancelAnimationFrame(animationRef.current);
    }
    setIsMonitoring(false);
  };

  const processVideo = () => {
    const video = videoRef.current;
    const canvas = canvasRef.current;
    
    if (!video || !canvas) return;

    const ctx = canvas.getContext('2d');
    canvas.width = video.videoWidth;
    canvas.height = video.videoHeight;

    const detect = () => {
      if (!isMonitoring) return;
      
      ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
      
      // Simulated face detection overlay
      ctx.strokeStyle = '#00ff00';
      ctx.lineWidth = 2;
      const faceX = canvas.width * 0.35;
      const faceY = canvas.height * 0.25;
      const faceWidth = canvas.width * 0.3;
      const faceHeight = canvas.height * 0.4;
      ctx.strokeRect(faceX, faceY, faceWidth, faceHeight);
      
      // Eye tracking points
      ctx.fillStyle = '#00ff00';
      ctx.beginPath();
      ctx.arc(faceX + faceWidth * 0.3, faceY + faceHeight * 0.3, 3, 0, Math.PI * 2);
      ctx.arc(faceX + faceWidth * 0.7, faceY + faceHeight * 0.3, 3, 0, Math.PI * 2);
      ctx.fill();
      
      animationRef.current = requestAnimationFrame(detect);
    };

    detect();
  };

  const simulateBehaviorAnalysis = () => {
    const interval = setInterval(() => {
      if (!isMonitoring) {
        clearInterval(interval);
        return;
      }

      // Simulate behavior metrics
      const newMetrics = {
        headPose: Math.random() * 100,
        eyeGaze: Math.random() * 100,
        handMovement: Math.random() * 100,
        voiceDetection: Math.random() * 30,
        phoneDetection: Math.random() * 20,
        multiplePersons: Math.random() * 15
      };

      setBehaviorMetrics(newMetrics);

      // Check for suspicious behavior
      const anomalyScore = (
        (newMetrics.headPose > 75 ? 1 : 0) +
        (newMetrics.eyeGaze > 80 ? 1 : 0) +
        (newMetrics.handMovement > 85 ? 1 : 0) +
        (newMetrics.voiceDetection > 20 ? 1 : 0) +
        (newMetrics.phoneDetection > 15 ? 1 : 0) +
        (newMetrics.multiplePersons > 10 ? 1 : 0)
      );

      if (anomalyScore >= 2) {
        const behaviors = [];
        if (newMetrics.headPose > 75) behaviors.push('Excessive head movement');
        if (newMetrics.eyeGaze > 80) behaviors.push('Looking away from screen');
        if (newMetrics.handMovement > 85) behaviors.push('Unusual hand gestures');
        if (newMetrics.voiceDetection > 20) behaviors.push('Voice detected');
        if (newMetrics.phoneDetection > 15) behaviors.push('Phone detected');
        if (newMetrics.multiplePersons > 10) behaviors.push('Multiple persons detected');
        
        addAlert(behaviors.join(', '), 'warning');
        setStats(prev => ({
          ...prev,
          suspiciousActions: prev.suspiciousActions + 1,
          alertLevel: anomalyScore >= 3 ? 'high' : 'medium'
        }));
      }
    }, 3000);

    return () => clearInterval(interval);
  };

  const addAlert = (message, type) => {
    const newAlert = {
      id: Date.now(),
      message,
      type,
      timestamp: new Date().toLocaleTimeString()
    };
    setAlerts(prev => [newAlert, ...prev].slice(0, 5));
  };

  const getAlertColor = () => {
    switch(stats.alertLevel) {
      case 'high': return 'bg-red-500';
      case 'medium': return 'bg-yellow-500';
      default: return 'bg-green-500';
    }
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-blue-900 to-slate-900 text-white p-6">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="text-center mb-8">
          <h1 className="text-4xl font-bold mb-2 bg-gradient-to-r from-blue-400 to-cyan-400 bg-clip-text text-transparent">
            THE SILENT INVIGILATOR
          </h1>
          <p className="text-gray-400">Autonomous Real-Time Exam Malpractice Detection System</p>
        </div>

        {/* Stats Dashboard */}
        <div className="grid grid-cols-1 md:grid-cols-3 gap-4 mb-6">
          <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-slate-700">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-gray-400 text-sm">Alert Level</p>
                <p className="text-2xl font-bold capitalize">{stats.alertLevel}</p>
              </div>
              <div className={`w-3 h-3 rounded-full ${getAlertColor()} animate-pulse`}></div>
            </div>
          </div>
          
          <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-slate-700">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-gray-400 text-sm">Suspicious Actions</p>
                <p className="text-2xl font-bold">{stats.suspiciousActions}</p>
              </div>
              <AlertTriangle className="text-yellow-500" />
            </div>
          </div>
          
          <div className="bg-slate-800/50 backdrop-blur p-4 rounded-lg border border-slate-700">
            <div className="flex items-center justify-between">
              <div>
                <p className="text-gray-400 text-sm">Students Monitored</p>
                <p className="text-2xl font-bold">{stats.totalStudents}</p>
              </div>
              <Users className="text-blue-500" />
            </div>
          </div>
        </div>

        {/* Main Content */}
        <div className="grid grid-cols-1 lg:grid-cols-3 gap-6">
          {/* Video Feed */}
          <div className="lg:col-span-2">
            <div className="bg-slate-800/50 backdrop-blur rounded-lg border border-slate-700 p-4">
              <div className="flex items-center justify-between mb-4">
                <h2 className="text-xl font-semibold flex items-center gap-2">
                  <Camera className="text-blue-400" />
                  Live Video Feed
                </h2>
                <button
                  onClick={isMonitoring ? stopMonitoring : startMonitoring}
                  className={`px-4 py-2 rounded-lg font-medium transition-colors ${
                    isMonitoring 
                      ? 'bg-red-500 hover:bg-red-600' 
                      : 'bg-blue-500 hover:bg-blue-600'
                  }`}
                >
                  {isMonitoring ? 'Stop Monitoring' : 'Start Monitoring'}
                </button>
              </div>
              
              <div className="relative bg-black rounded-lg overflow-hidden aspect-video">
                <video 
                  ref={videoRef}
                  autoPlay 
                  playsInline
                  muted
                  className="absolute inset-0 w-full h-full object-cover"
                />
                <canvas 
                  ref={canvasRef}
                  className="absolute inset-0 w-full h-full"
                />
                {!isMonitoring && (
                  <div className="absolute inset-0 flex items-center justify-center bg-slate-900/80">
                    <div className="text-center">
                      <Camera className="w-16 h-16 mx-auto mb-4 text-gray-600" />
                      <p className="text-gray-400">Click "Start Monitoring" to begin</p>
                    </div>
                  </div>
                )}
              </div>
            </div>

            {/* Behavior Metrics */}
            <div className="bg-slate-800/50 backdrop-blur rounded-lg border border-slate-700 p-4 mt-4">
              <h3 className="text-lg font-semibold mb-4 flex items-center gap-2">
                <Activity className="text-cyan-400" />
                Behavior Analysis Metrics
              </h3>
              <div className="space-y-3">
                {Object.entries(behaviorMetrics).map(([key, value]) => (
                  <div key={key}>
                    <div className="flex justify-between text-sm mb-1">
                      <span className="capitalize">{key.replace(/([A-Z])/g, ' $1').trim()}</span>
                      <span className={value > 70 ? 'text-red-400' : 'text-green-400'}>
                        {value.toFixed(1)}%
                      </span>
                    </div>
                    <div className="w-full bg-slate-700 rounded-full h-2">
                      <div 
                        className={`h-2 rounded-full transition-all duration-500 ${
                          value > 70 ? 'bg-red-500' : value > 50 ? 'bg-yellow-500' : 'bg-green-500'
                        }`}
                        style={{ width: `${value}%` }}
                      />
                    </div>
                  </div>
                ))}
              </div>
            </div>
          </div>

          {/* Alerts Panel */}
          <div className="lg:col-span-1">
            <div className="bg-slate-800/50 backdrop-blur rounded-lg border border-slate-700 p-4">
              <h2 className="text-xl font-semibold mb-4 flex items-center gap-2">
                <Eye className="text-yellow-400" />
                Real-Time Alerts
              </h2>
              
              <div className="space-y-3 max-h-[600px] overflow-y-auto">
                {alerts.length === 0 ? (
                  <div className="text-center py-8 text-gray-500">
                    <CheckCircle className="w-12 h-12 mx-auto mb-2" />
                    <p>No alerts detected</p>
                    <p className="text-sm">System monitoring normally</p>
                  </div>
                ) : (
                  alerts.map(alert => (
                    <div 
                      key={alert.id}
                      className={`p-3 rounded-lg border ${
                        alert.type === 'error' 
                          ? 'bg-red-900/20 border-red-500' 
                          : 'bg-yellow-900/20 border-yellow-500'
                      }`}
                    >
                      <div className="flex items-start gap-2">
                        <AlertTriangle className="w-5 h-5 flex-shrink-0 mt-0.5" />
                        <div className="flex-1 min-w-0">
                          <p className="text-sm font-medium">{alert.message}</p>
                          <p className="text-xs text-gray-400 mt-1">{alert.timestamp}</p>
                        </div>
                      </div>
                    </div>
                  ))
                )}
              </div>
            </div>

            {/* Algorithm Info */}
            <div className="bg-slate-800/50 backdrop-blur rounded-lg border border-slate-700 p-4 mt-4">
              <h3 className="text-sm font-semibold mb-2 text-blue-400">Detection Methods</h3>
              <ul className="text-xs space-y-1 text-gray-400">
                <li>• Face & Pose Detection (MediaPipe)</li>
                <li>• Eye Gaze Tracking</li>
                <li>• Head Pose Estimation</li>
                <li>• Object Detection (Phone/Papers)</li>
                <li>• Multi-Person Detection</li>
                <li>• Audio Analysis</li>
                <li>• Temporal Anomaly Scoring</li>
              </ul>
            </div>
          </div>
        </div>
      </div>
    </div>
  );
};

export default SilentInvigilator;
